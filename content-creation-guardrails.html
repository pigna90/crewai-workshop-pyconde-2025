Content Creation Guardrails

Task Overview

In this assignment, you'll implement guardrails for AI-generated content to ensure it meets quality and safety standards. You'll learn how to validate, filter, and moderate AI outputs effectively.

class ContentGuardrail:
    def __init__(self, api_key):
        self.api_key = api_key
        openai.api_key = api_key

    def validate_content(self, content):
        # Check for inappropriate content
        if self._contains_inappropriate_content(content):
            return False, "Content contains inappropriate material"
        
        # Check for quality standards
        if not self._meets_quality_standards(content):
            return False, "Content does not meet quality standards"
        
        return True, "Content approved"

    def _contains_inappropriate_content(self, content):
        # Implement content moderation logic
        inappropriate_keywords = ["hate", "violence", "explicit"]
        return any(keyword in content.lower() for keyword in inappropriate_keywords)

    def _meets_quality_standards(self, content):
        # Implement quality check logic
        min_length = 100
        return len(content) >= min_length

# Example usage
if __name__ == "__main__":
    guardrail = ContentGuardrail("your-api-key-here")
    content = "This is a sample content that meets all standards."
    is_valid, message = guardrail.validate_content(content)
    print(f"Validation result: {is_valid}")
    print(f"Message: {message}")
